``` csharp
class Demo
{
  public int Id { get; set; }
  public string Name { get; set; }
}
static void Main(string[] args)
{
  List<Demo> data = new List<Demo>();
  string[] names = (from x in data select x.Name).ToArray();
  //或者
  string[] names = data.Select(x=>x.Name).ToArray();
}
```
